The percentage covered by the confidence interval for 1/lambda: X1.96 * (S/n) is

ul <- mean(AllMeans) +  1.96 * real_sample_mean_sd
ll <- mean(AllMeans) -  1.96 * real_sample_mean_sd
inside <- 0
for (i in AllMeans) {
    if (i > ll & i < ul){
        inside <- inside + 1
    }
}
coverage <- inside/length(AllMeans)
coverage
## [1] 0.952
